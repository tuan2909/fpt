CREATE TABLE VANPHONG
(
    MAVANPHONG VARCHAR(50) NOT NULL,
    DIADIEM VARCHAR(50) NULL,
    CONSTRAINT PK_VANPHONG PRIMARY KEY(MAVANPHONG)
);

CREATE TABLE NHANVIEN
(
	MANHANVIEN VARCHAR(50) NOT NULL,
    TEN VARCHAR(50) NULL,
    CONSTRAINT PK_NHANVIEN PRIMARY KEY(MANHANVIEN)
);

CREATE TABLE THANNHAN
(
	MATHANNHAN VARCHAR(50) NOT NULL,
    TEN VARCHAR(50) NULL,
    MOIQUANHE VARCHAR(50) NULL,
    NGAYSINH DATETIME(50) NULL,
    CONSTRAINT PK_THANHAN PRIMARY KEY(MATHANNHAN)
);
